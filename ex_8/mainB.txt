#include <stdio.h>
#include <signal.h>
#include <unistd.h>
#include <native/task.h>
#include <native/sem.h>
#include <native/mutex.h>
#include <native/timer.h>
#include <sys/mman.h>
#include <pthread.h>
#include <sched.h>
#include <rtdk.h>

#define CORE_NUM 1
#define time_step = 1000

RT_SEM sync_sem;
RT_SEM res_sem;



void busy_wait_us(unsigned long delay){
	for(; delay > 0; delay--){
		rt_timer_spin(1000);
	}
}


void low_func(void *arg){
	unsigned long duration = 30000000000;
	unsigned long endTime = rt_timer_read() + duration;

	rt_printf("Task %s waiting for sync\n", arg);	
	rt_sem_p(&sync_sem, TM_INFINITE);	
	rt_printf("Task %s got sync\n", arg);

	rt_printf("Task %s Waiting for semaphore\n", arg);
	rt_sem_p(&res_sem, TM_INFINITE);
	rt_printf("Task %s got semaphore\n", arg);
	

	rt_printf("Task %s busy wait\n", arg);	
	busy_wait_us(30);

	rt_printf("Task %s about to release semaphore\n", arg);
	rt_sem_v(&res_sem);
	
	
	rt_printf("Task %s done\n", arg);

	if (rt_timer_read() > endTime){
		rt_printf("Time expired\n");
		rt_task_delete(NULL);		
	}
	if (rt_task_yield()){
		rt_printf("No yield\n");
		rt_task_delete(NULL);
	}

	return;
}

void med_func(void *arg){
	unsigned long duration = 30000000000;
	unsigned long endTime = rt_timer_read() + duration;
	
	rt_printf("Task %s waiting for sync\n", arg);	
	rt_sem_p(&sync_sem, TM_INFINITE);
	rt_printf("Task %s got sync\n", arg);

	rt_printf("Task %s sleep\n", arg);
	rt_task_sleep(10*1000);

	rt_printf("Task %s busy wait start\n", arg);
	busy_wait_us(50);
	rt_printf("Task %s busy wait done\n", arg);

	if (rt_timer_read() > endTime){
		rt_printf("Time expired\n");
		rt_task_delete(NULL);		
	}
	if (rt_task_yield()){
		rt_printf("No yield\n");
		rt_task_delete(NULL);
	}

	return;
}

void high_func(void *arg){
	unsigned long duration = 30000000000;
	unsigned long endTime = rt_timer_read() + duration;

	rt_printf("Task %s waiting for sync\n", arg);	
	rt_sem_p(&sync_sem, TM_INFINITE);
	rt_printf("Task %s got sync\n", arg);

	rt_printf("Task %s sleep\n", arg);
	rt_task_sleep(20*1000);

	rt_printf("Task %s Waiting for semaphore\n", arg);
	rt_sem_p(&res_sem, TM_INFINITE);
	rt_printf("Task %s got semaphore\n", arg);


	rt_printf("Task %s busy wait\n", arg);	
	busy_wait_us(20);

	rt_sem_v(&res_sem);
	rt_printf("Task %s release semaphore\n", arg);

	rt_printf("Task %s done\n", arg);	

	if (rt_timer_read() > endTime){
		rt_printf("Time expired\n");
		rt_task_delete(NULL);		
	}
	if (rt_task_yield()){
		rt_printf("No yield\n");
		rt_task_delete(NULL);
	}

	return;
}


int main(int argc, char* argv[]){
	mlockall(MCL_CURRENT|MCL_FUTURE);

	rt_print_auto_init(1);
	rt_task_shadow(NULL, "MainTask", 80, T_CPU(CORE_NUM));

	// Create semaphore
	rt_sem_create(&sync_sem, NULL, 0, S_PRIO);
	rt_sem_create(&res_sem, NULL, 1, S_FIFO); 
	
	// Create tasks
	RT_TASK task_L;
	RT_TASK task_M;
	RT_TASK task_H;		
	rt_task_create(&task_L, NULL, 0, 20, T_CPU(CORE_NUM));
	rt_task_create(&task_M, NULL, 0, 40, T_CPU(CORE_NUM));
	rt_task_create(&task_H, NULL, 0, 60, T_CPU(CORE_NUM));
	
	// Start tasks
	rt_task_start(&task_L, low_func, (void*)("Low"));
	rt_task_start(&task_M, med_func, (void*)("Med"));
	rt_task_start(&task_H, high_func, (void*)("High"));
	
	RTIME sleep_time = 100*1000*1000;
	rt_task_sleep(sleep_time);
	
	// Broadcast Semaphore
	rt_sem_broadcast(&sync_sem);
	rt_printf("start test\n");

	rt_task_sleep(sleep_time);
	rt_printf("end test\n");

	// Delete semaphore
	rt_sem_delete(&res_sem);
	rt_sem_delete(&sync_sem);
	rt_printf("semaphores deleted\n");

	return 0;
}
